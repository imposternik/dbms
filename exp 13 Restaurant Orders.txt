CREATE TABLE menu (
    item_id INT PRIMARY KEY,
    item_name VARCHAR(100) NOT NULL,
    price DECIMAL(6,2) CHECK (price > 0),
    category VARCHAR(50) NOT NULL
);

CREATE TABLE orders (
    order_id INT PRIMARY KEY,
    item_id INT,
    customer_name VARCHAR(100) NOT NULL,
    order_date DATE NOT NULL,
    quantity INT CHECK (quantity > 0),
    FOREIGN KEY (item_id) REFERENCES menu(item_id)
);

‚úÖ 5. Display Order Details with Menu Information
sql
Copy
Edit
SELECT 
    o.order_id,
    o.customer_name,
    m.item_name,
    m.category,
    o.quantity,
    m.price,
    o.order_date
FROM 
    orders o
JOIN 
    menu m ON o.item_id = m.item_id;
‚úÖ 6. Show Total Orders per Menu Item
sql
Copy
Edit
SELECT 
    m.item_id,
    m.item_name,
    SUM(o.quantity) AS total_quantity_ordered
FROM 
    menu m
LEFT JOIN 
    orders o ON m.item_id = o.item_id
GROUP BY 
    m.item_id, m.item_name;
