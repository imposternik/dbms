CREATE TABLE teams (
    team_id INT PRIMARY KEY,
    team_name VARCHAR(100) NOT NULL,
    city VARCHAR(50) NOT NULL
);

CREATE TABLE matches (
    match_id INT PRIMARY KEY,
    team1_id INT,
    team2_id INT,
    match_date DATE NOT NULL,
    venue VARCHAR(100) NOT NULL,
    winner_id INT,
    FOREIGN KEY (team1_id) REFERENCES teams(team_id),
    FOREIGN KEY (team2_id) REFERENCES teams(team_id),
    FOREIGN KEY (winner_id) REFERENCES teams(team_id)
);

âœ… 6. Show Total Matches Played by Each Team
sql
Copy
Edit
SELECT 
    t.team_id,
    t.team_name,
    COUNT(m.match_id) AS total_matches
FROM 
    teams t
LEFT JOIN matches m ON t.team_id = m.team1_id OR t.team_id = m.team2_id
GROUP BY 
    t.team_id, t.team_name;

DELIMITER $$

CREATE PROCEDURE get_matches_by_venue(IN venue_name VARCHAR(100))
BEGIN
    SELECT 
        m.match_id,
        t1.team_name AS team1,
        t2.team_name AS team2,
        m.match_date,
        m.venue,
        tw.team_name AS winner
    FROM 
        matches m
    JOIN teams t1 ON m.team1_id = t1.team_id
    JOIN teams t2 ON m.team2_id = t2.team_id
    JOIN teams tw ON m.winner_id = tw.team_id
    WHERE 
        m.venue = venue_name;
END $$

DELIMITER ;
